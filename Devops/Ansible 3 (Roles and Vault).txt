#ansible-playbook -i slaves.txt roles.yml

#ansible-playbook -i slaves.txt roles.yml --ask-vault-pass

#cat roles.yml

#ansible-vault view roles.yaml

FOR EDITING
#ansible-vault edit roles.yaml

FOR CHANGING PASSWORD
#ansible-vault rekey roles.yaml

FOR Totally remove the PASSWORD
#ansible-vault decrypt roles.yaml
