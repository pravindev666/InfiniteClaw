# Use an official Ubuntu base image
FROM ubuntu:latest

# Set environment variables to non-interactive for automated installs
ENV DEBIAN_FRONTEND=noninteractive

# Update package list and install Vim
RUN apt-get update && apt-get install -y vim

# Update and install required packages
RUN apt-get update && \
    apt-get install -y wget unzip && \
    rm -rf /var/lib/apt/lists/*

# Download Terraform
RUN wget https://releases.hashicorp.com/terraform/1.8.4/terraform_1.8.4_linux_amd64.zip

# List the directory contents (this will be in the image build logs)
RUN ls -la

# Unzip the Terraform package
RUN unzip terraform_1.8.4_linux_amd64.zip

# List the directory contents again (this will be in the image build logs)
RUN ls -la

# Move the Terraform binary to /usr/local/bin
RUN mv terraform /usr/local/bin/

# Verify the Terraform installation
RUN terraform --version
